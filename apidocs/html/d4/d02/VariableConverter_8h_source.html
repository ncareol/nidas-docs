318 /* static */
319 inline double Polynomial::eval(double x,float *p, unsigned int np)
320 {
321  double y = 0.0;
322  if (np == 0) return y;
323  for (unsigned int i = np - 1; i > 0; i--) {
324  y += p[i];
325  y *= x;
326  }
327  y += p[0];
328  return y;
329 }
