691 /*
692  * Does a listen and then waits for the connection.
693  */
694 template<class SocketT>
695 SocketT* McSocket<SocketT>::accept(Inet4PacketInfoX& pktinfo)
696 {
697  listen();
698  _connectCond.lock();
699  while(!_socketOffered) _connectCond.wait();
700  SocketT* socket = _newsocket;
701  pktinfo = _newpktinfo;
702  _newsocket = 0;
703  _connectCond.unlock();
704  VLOG(("accept offerErrno=%d", _offerErrno));
705  if (!socket) throw IOException("McSocket","accept",_offerErrno);
706  return socket;

751 template<class SocketT>
752 void McSocket<SocketT>::joinMulticaster()
753 {
754  // can't do a multicaster->join here, that would deadlock
755  // because we'd be waiting for ourselves,
756  // so we spawn a detached thread to join and delete it.
757 
758  _multicaster_mutex.lock();
759 
760  if (_multicaster) {
761  VLOG(("Mcsocket::offer creating/starting joiner"));
762  ThreadJoiner* joiner = new ThreadJoiner(_multicaster);
763  _multicaster = 0;
764  try {
765  joiner->start();
766  }
767  // shouldn't happen
768  catch(const Exception& e) {
769  Logger::getInstance()->log(LOG_ERR,"%s",e.what());
770  }
771  VLOG(("Mcsocket::offer joiner started"));
772  }
773  VLOG(("Mcsocket::offer doing connectCond.lock"));
774  _multicaster_mutex.unlock();
775 }
776 /*
777  * Method that executes in the thread of the McSocketListener
778  * or the McSocketMulticaster, and notifies whoever did
779  * the listen() or request() that the socket is connected.
780  */
781 template<class SocketT>
782 void McSocket<SocketT>::offer(SocketT* socket,const Inet4PacketInfoX& pktinfo)
783 {
784  joinMulticaster();
785 
786  if (socket) connected(socket,pktinfo);
787 
788  _connectCond.lock();
789  _socketOffered = true;
790  _offerErrno = 0;
791  _newsocket = socket;
792  _newpktinfo = pktinfo;
793  _connectCond.signal();
794  _connectCond.unlock();
795 
796  // note: don't access any class variables or virtual methods
797  // after the above unlock. The calling thread may have
798  // deleted this object after it received the connected Socket.
799 }

894 template<class SocketT>
895 int McSocketMulticaster<SocketT>::run()
896 {
897  MulticastSocket* requestmsock = 0;
898 
899  int sockfd = (_serverSocket ? _serverSocket->getFd() :
900  _datagramSocket->getFd());
901 
902 #ifdef HAVE_PPOLL
903  struct pollfd fds;
904  fds.fd = sockfd;
905 #ifdef POLLRDHUP
906  fds.events = POLLIN | POLLRDHUP;
907 #else
908  fds.events = POLLIN;
909 #endif
910 #else
911  fd_set fdset;
912  FD_ZERO(&fdset);
913 #endif
914 
915  // wait for this amount of time after a datagram send to receive a
916  // connection or receipt of UDP data
917  struct timespec timeout;
918  timeout.tv_sec = 0;
919  timeout.tv_nsec = NSECS_PER_SEC / 4; // portion of a second
920 
921  // get the existing signal mask
922  sigset_t sigmask;
923  pthread_sigmask(SIG_BLOCK,NULL,&sigmask);
924  // unblock SIGUSR1 in pselect/ppoll
925  sigdelset(&sigmask,SIGUSR1);
926 
927  Inet4SocketAddress mcsockaddr =
928  _mcsocket->getInet4McastSocketAddress();
929  Inet4Address mcaddr = mcsockaddr.getInet4Address();
930 
931  std::vector<Inet4NetworkInterface> ifaces;
932 
933  if (mcaddr.isMultiCastAddress())
934  {
935  VLOG(("") << mcsockaddr.toString() << " is a multicast address.");
936  _requestSocket = requestmsock = new MulticastSocket();
937  if (_mcsocket->getInterface().getAddress() == Inet4Address(INADDR_ANY))
938  {
939  listMulticastInterfaces(requestmsock, ifaces);
940  }
941  }
942  else
943  {
944  VLOG(("") << mcsockaddr.toString()
945  << " is not a multicast address, using a datagram request socket.");
946  _requestSocket = new DatagramSocket();
947  }
948 
949  McSocketDatagram dgram;
950  dgram.setMagic(dgram.magicVal);
951  dgram.setSocketAddress(mcsockaddr);
952  dgram.setRequestType(_mcsocket->getRequestType());
953  dgram.setSocketType(getMcSocketType(_mcsocket));
954  if (_serverSocket)
955  dgram.setRequesterListenPort(_serverSocket->getLocalPort());
956  else
957  dgram.setRequesterListenPort(_datagramSocket->getLocalPort());
958 
959  for (int numCasts=0; !isInterrupted() ; numCasts++) {
960  // If multicast, loop over interfaces
961  try {
962  if (requestmsock && ifaces.size() > 0) {
963  Inet4NetworkInterface iface = ifaces[numCasts % ifaces.size()];
964  requestmsock->setInterface(mcaddr,iface);
965  _requestSocket->send(dgram);
966  }
967  else {
968  _requestSocket->send(dgram);
969  }
970  if (!(numCasts % 300)) {
971  ILOG(("") << "sent " << numCasts << " dgrams" <<
972  ", requestType=" << dgram.getRequestType() <<
973  ", port=" << dgram.getRequesterListenPort() <<
974  ", socketType=" << dgram.getSocketType() <<
975  ", len=" << dgram.getLength() <<
976  ", #mcifaces=" << ifaces.size());
977  }
978  }
979  catch(const IOException& e)
980  {
981  // perhaps the interface has disappeared. Log the error.
982  WLOG(("McSocketMulticaster: ")
983  << _requestSocket->getLocalSocketAddress().toString()
984  << ": " << e.what());
985  sleep(10);
986  if (requestmsock)
987  {
988  _requestSocket->close();
989  delete _requestSocket;
990  // close and re-create the socket
991  _requestSocket = requestmsock = new MulticastSocket();
992  // re-create the list of interfaces.
993  listMulticastInterfaces(requestmsock, ifaces);
994  }
995  continue;
996  }
997 
998  int res;
999 #ifdef HAVE_PPOLL
1000  res = ::ppoll(&fds,1,&timeout,&sigmask);
1001 
1002  if (res > 0) {
1003 
1004  if (fds.revents & POLLERR) {
1005  _mcsocketMutex.lock();
1006  if (_mcsocket) _mcsocket->offer(errno);
1007  _mcsocketMutex.unlock();
1008  if (_serverSocket) _serverSocket->close(); // OK to close twice
1009  if (_datagramSocket) _datagramSocket->close();
1010  _requestSocket->close();
1011  throw IOException("McSocket","ppoll",errno);
1012  }
1013 
1014 #ifdef POLLRDHUP
1015  if (fds.revents & (POLLHUP | POLLRDHUP))
1016 #else
1017  if (fds.revents & POLLHUP)
1018 #endif
1019  WLOG(("%s POLLHUP","McSocket"));
1020 
1021  }
1022 
1023 #else
1024  assert(sockfd >= 0 && sockfd < FD_SETSIZE); // FD_SETSIZE=1024
1025  FD_SET(sockfd, &fdset);
1026  res = ::pselect(sockfd+1,&fdset,0,0,&timeout,&sigmask);
1027 #endif
1028  if (res <= 0) {
1029  if (res == 0) continue;
1030  if (errno == EINTR) break;
1031  _mcsocketMutex.lock();
1032  if (_mcsocket) _mcsocket->offer(errno);
1033  _mcsocketMutex.unlock();
1034  if (_serverSocket) _serverSocket->close(); // OK to close twice
1035  if (_datagramSocket) _datagramSocket->close();
1036  _requestSocket->close();
1037  throw IOException("McSocket","poll/select",errno);
1038  }
1039 
1040  if (_serverSocket) {
1041  Socket* socket = _serverSocket->accept();
1042  DLOG(("accepted socket connection from ") <<
1043  socket->getRemoteSocketAddress().toString() << " on " <<
1044  _serverSocket->getLocalSocketAddress().toString());
1045  _serverSocket->close();
1046  nidas::util::Inet4PacketInfoX pktinfo;
1047  if (socket->getRemoteSocketAddress().getFamily() == AF_INET) {
1048  nidas::util::Inet4SocketAddress remoteAddr =
1049  nidas::util::Inet4SocketAddress((const struct sockaddr_in*)
1050  socket->getRemoteSocketAddress().getConstSockAddrPtr());
1051  pktinfo.setRemoteSocketAddress(remoteAddr);
1052  }
1053  if (socket->getLocalSocketAddress().getFamily() == AF_INET) {
1054  nidas::util::Inet4SocketAddress localAddr =
1055  nidas::util::Inet4SocketAddress((const struct sockaddr_in*)
1056  socket->getLocalSocketAddress().getConstSockAddrPtr());
1057  pktinfo.setLocalAddress(localAddr.getInet4Address());
1058  pktinfo.setDestinationAddress(localAddr.getInet4Address());
1059  }
1060  _mcsocketMutex.lock();
1061  if (_mcsocket) _mcsocket->offer((SocketT*)socket,pktinfo);
1062  _mcsocketMutex.unlock();
1063  }
1064  else {
1065  // If fishing for UDP responses, send out at least 3 multicasts
1066  // to see if we get more than one response.
1067  if (numCasts < 3) continue;
1068 
1069  // We know there is data at the socket, do a MSG_PEEK to get
1070  // information about the first packet.
1071  nidas::util::Inet4PacketInfoX pktinfo;
1072  _datagramSocket->receive(dgram,pktinfo,MSG_PEEK);
1073  if (dgram.getSocketAddress().getFamily() == AF_INET) {
1074  nidas::util::Inet4SocketAddress remoteAddr =
1075  nidas::util::Inet4SocketAddress((const struct sockaddr_in*)
1076  dgram.getSocketAddress().getConstSockAddrPtr());
1077  pktinfo.setRemoteSocketAddress(remoteAddr);
1078  }
1079 
1080  _mcsocketMutex.lock();
1081  if (_mcsocket) _mcsocket->offer((SocketT*)_datagramSocket,pktinfo);
1082  _datagramSocket = 0; // we no longer own _datagramSocket
1083  _mcsocketMutex.unlock();
