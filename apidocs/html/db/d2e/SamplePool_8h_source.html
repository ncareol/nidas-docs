217 template<class SampleType>
218  SamplePool<SampleType>::SamplePool():
219  _smallSamples(0), _mediumSamples(0), _largeSamples(0),
220  _smallSize(0), _mediumSize(0), _largeSize(0),
221  _poolLock(),
222  _nsmall(0),_nmedium(0),_nlarge(0),_nsamplesOut(0),_nsamplesAlloc(0)
223 {
224  // Initial size of pool of small samples around 16K bytes
225  _smallSize = 16384 / (sizeof(SampleType) + SMALL_SAMPLE_MAXSIZE * SampleType::sizeofDataType());
226  // When we expand the size of the pool, we expand by 50%
227  // so minimum size should be at least 2.
228  if (_smallSize < 2) _smallSize = 2;
229 
230  _mediumSize = _smallSize / (MEDIUM_SAMPLE_MAXSIZE / SMALL_SAMPLE_MAXSIZE);
231  if (_mediumSize < 2) _mediumSize = 2;
232 
233  _largeSize = _mediumSize / 2;
234  if (_largeSize < 2) _largeSize = 2;
235 
236  _smallSamples = new SampleType*[_smallSize];
237  _mediumSamples = new SampleType*[_mediumSize];
238  _largeSamples = new SampleType*[_largeSize];
239 #ifdef DEBUG
240  DLOG(("nsmall=%d, nmedium=%d, nlarge=%d",_smallSize,_mediumSize, _largeSize));
241 #endif
242 }

256 template<class SampleType>
257 SampleType* SamplePool<SampleType>::getSample(unsigned int len)
258 {
259 
260  nidas::util::Synchronized pooler(_poolLock);
261 
262  // Shouldn't get back more than I've dealt out
263  // If we do, that's an indication that reference counting
264  // is screwed up.
265 
266  assert(_nsamplesOut >= 0);
267 
268  // conservation of sample numbers:
269  // total number allocated must equal:
270  // number in the pool (_nsmall + _nmedium + _nlarge), plus
271  // number held by others.
272  assert(_nsamplesAlloc == _nsmall + _nmedium + _nlarge + _nsamplesOut);
273 
274  // get a sample from the appropriate pool, unless is is empty
275  // and there are 2 or more available from the next larger pool.
276  if (len < SMALL_SAMPLE_MAXSIZE && (_nsmall > 0 ||
277  (_nmedium + _nlarge) < 4))
278  return getSample((SampleType**)_smallSamples,&_nsmall,len);
279  else if (len < MEDIUM_SAMPLE_MAXSIZE && (_nmedium > 0 || _nlarge < 2))
280  return getSample((SampleType**)_mediumSamples,&_nmedium,len);
281  else return getSample((SampleType**)_largeSamples,&_nlarge,len);
282 }

284 template<class SampleType>
285 SampleType* SamplePool<SampleType>::getSample(SampleType** vec,
286  int *n, unsigned int len)
287 {
288 
289  SampleType *sample;
290 #ifdef DEBUG
291  std::cerr << "getSample, this=" << std::hex << this <<
292  " pool=" << vec << std::dec <<
293  " *n=" << *n << std::endl;
294 #endif
295  int i = *n - 1;
296 
297  if (i >= 0) {
298  sample = vec[i];
299  if (sample->getAllocLength() < len) sample->allocateData(len);
300 #ifndef NDEBUG
301  else if (sample->getAllocLength() > len) {
302  // If the sample has been previously allocated, and its length
303  // is at least one more than we need, set the one-past-the-end
304  // data value to a noticable value. Then if a buggy process method
305  // reads past the end of a sample, they'll get a value that should
306  // raise questions about the results, rather than something
307  // that might go unnoticed.
308 
309  // valgrind won't complain in these situations unless one reads
310  // past the allocated size.
311 
312  // For character data (sizeof(T) == 1), we'll use up to
313  // 4 '\x80's as the weird value.
314  // For larger sizes, we'll use floatNAN. This will convert to
315  // 0 for integer samples.
316 
317  extern const float floatNAN;
318 
319  if (sample->sizeofDataType() == 1) {
320  static const char weird[4] = { '\x80','\x80','\x80','\x80' };
321  int nb = std::min(4U,sample->getAllocLength()-len);
322  memcpy((char*)sample->getVoidDataPtr()+len,weird,nb);
323  }
324  else sample->setDataValue(len,floatNAN); // NAN converted to the data type.
325  }
326 #endif
327  sample->setDataLength(len);
328  *n = i;
329  sample->holdReference();
330  _nsamplesOut++;
331  return sample;
332  }
333 
334  sample = new SampleType();
335  sample->allocateData(len);
336  sample->setDataLength(len);
337  _nsamplesAlloc++;
338  _nsamplesOut++;
339  return sample;
340 
341 }

343 template<class SampleType>
344 void SamplePool<SampleType>::putSample(const SampleType *sample) {
345 
346  nidas::util::Synchronized pooler(_poolLock);
347 
348  assert(_nsamplesOut >= 0);
349  assert(_nsamplesAlloc == _nsmall + _nmedium + _nlarge + _nsamplesOut);
350 
351  unsigned int len = sample->getAllocLength();
352  if (len < SMALL_SAMPLE_MAXSIZE) {
353 #ifdef DEBUG
354  DLOG(("put small sample, len=%d,bytelen=%d,n=%d,size=%d",len,sample->getAllocByteLength(),_nsmall,_smallSize));
355 #endif
356  putSample(sample,(SampleType***)&_smallSamples,&_nsmall,&_smallSize);
357  }
358  else if (len < MEDIUM_SAMPLE_MAXSIZE) {
359 #ifdef DEBUG
360  DLOG(("put medium sample, len=%d,bytelen=%d,n=%d,size=%d",len,sample->getAllocByteLength(),_nmedium,_mediumSize));
361 #endif
362  putSample(sample,(SampleType***)&_mediumSamples,&_nmedium,&_mediumSize);
363  }
364  else {
365 #ifdef DEBUG
366  DLOG(("put large sample, len=%d,bytelen=%d,n=%d,size=%d",len,sample->getAllocByteLength(),_nlarge,_largeSize));
367 #endif

374  SampleType ***vec,int *n, int *nalloc)

376 
377  // increase by 50%
378  if (*n == *nalloc) {
379  // cerr << "reallocing, n=" << *n << " nalloc=" << *nalloc << endl;
380  // increase by 50%
381  int newalloc = *nalloc + (*nalloc >> 1);
382 #ifdef DEBUG

387  delete [] *vec;
388  *vec = newvec;
389  *nalloc = newalloc;
390  }
