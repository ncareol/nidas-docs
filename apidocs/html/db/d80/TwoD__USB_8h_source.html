389 stream_histogram(T& out, V* sizedist, unsigned int nbins)
390 {
391  bool zeros = true;
392  for (unsigned int i = 0; i < nbins; ++i)
393  {
394  if (sizedist[i] == 0)
395  {
396  out << ".";
397  zeros = true;
398  }
399  else if (zeros)
400  {
401  out << sizedist[i];
402  zeros = false;
403  }
404  else
405  {
406  out << ",";
407  out << sizedist[i];
408  }
409  }
